Spring Boot v2.5.3整合MyBatis

2021年8月08日
阅读时长: 2 分钟
  1. 打开IDEA单击 “File”，选择 “New Project”，切换到 “Spring Initializr” 选项卡下，根据安装的JDK选择对应的版本。单击 “Next” 开始下一步，选择项目所需的依赖，点击 “Finish” 完成。 转到pom.xml添加json的支持。
<dependency>
	<groupId>com.alibaba</groupId>
	<artifactId>fastjson</artifactId>
	<version>1.2.47</version>
</dependency>

  1. 创建数据库，修改 resources\application.properties，配置数据源和MyBatis。
# 创建数据库名为person，添加student表，包含id,name和sex属性
CREATE DATABASE;
USE `person`;
DROP TABLE IF EXISTS `student`;
CREATE TABLE `student` (
  `id` int(16) NOT NULL AUTO_INCREMENT COMMENT 'id',
  `name` varchar(32) COLLATE utf8_croatian_ci DEFAULT NULL COMMENT '姓名',
  `sex` tinyint(1) DEFAULT '1' COMMENT '性别',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_croatian_ci;

#数据源
spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver
spring.datasource.url=jdbc:mysql://localhost:3306/数据库名?useUnicode=true&characterEncoding=utf-8
spring.datasource.username=数据库用户名
spring.datasource.password=数据库密码
#mybatis
mybatis.mapperLocations=classpath:mapper/*.xml
  1. 要将 MyBatis 与 Spring 一起使用，至少需要SqlSessionFactory一个映射器接口。默认情况下，MyBatis-Spring-Boot-Starter 将搜索标有@Mapper注解的映射器。

MyBatis-Spring-Boot-Starter 将：

  • 自动检测现有 DataSource
  • 将创建并注册一个SqlSessionFactory传递实例DataSource作为输入使用SqlSessionFactoryBean
  • 将创建并注册的一个实例SqlSessionTemplate得到的出SqlSessionFactory
  • 自动扫描您的映射器，将它们链接到SqlSessionTemplate并将它们注册到 Spring 上下文，以便将它们注入到您的 bean 中。

在 resources 下新建mapper命名为 mapper/StudentMapper.xml。在 DemoApplication.java 同级目录下新建映射器 StudentMapper.java，业务处理 StudentService.java，控制层 StudentController.java 以及JavaBean。

StudentMapper.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
  PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
  "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<!--namespace必须为全类名 -->
<mapper namespace="com.example.demo.StudentMapper">

	<select id="query" resultType="com.example.demo.Student">
		select * from
		`student`
	</select>

	<insert id="insert" parameterType="com.example.demo.Student"
		useGeneratedKeys="true" keyProperty="id">
		INSERT INTO `student`
		(`id`,
		`name`, `sex`)
		VALUES(#{id}, #{name}, #{sex})
	</insert>

	<delete id="delete" parameterType="int">
		delete from `student` where
		id=#{id}
	</delete>

	<update id="update" parameterType="hashmap">
		update `student` set `name`=#{name},`sex`=#{sex}
		where id=#{id}
	</update>
	
	<select id="selectID" parameterType="int" resultType="com.example.demo.Student">
		select * from `student` where id=#{id}
	</select>
	
	<select id="selectName" parameterType="String" resultType="com.example.demo.Student">
		SELECT * FROM `student`
		WHERE `name` LIKE #{name}
	</select>

</mapper>

StudentMapper.java

package com.example.demo;

import org.apache.ibatis.annotations.Mapper;

import java.util.HashMap;
import java.util.List;

@Mapper
public interface StudentMapper
{
	List<Student> query();
	
	int insert(Student s);
	
	int delete(int id);
	
	int update(HashMap<String,Object> m);
	
	List<Student> selectID(int id);
	
	List<Student> selectName(String name);

}

StudentService.java

package com.example.demo;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.stereotype.Service;

import java.util.HashMap;
import java.util.List;

@Service
public class StudentService implements StudentMapper
{
	@Qualifier("studentMapper")
	@Autowired
	private StudentMapper stuMapper;

	@Override
	public List<Student> query()
	{
		return stuMapper.query();
	}

	@Override
	public int insert(Student s)
	{
		return stuMapper.insert(s);
	}

	@Override
	public int delete(int id)
	{
		return stuMapper.delete(id);
	}

	@Override
	public int update(HashMap<String, Object> m)
	{
		return stuMapper.update(m);
	}

	@Override
	public  List<Student> selectID(int id)
	{
		return stuMapper.selectID(id);
	}

	@Override
	public List<Student> selectName(String name)
	{
		return stuMapper.selectName(name);
	}

}

StudentController.java

package com.example.demo;

import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.servlet.View;

import java.util.HashMap;
import java.util.List;

@Controller
public class StudentController
{

	@Autowired
	@Qualifier("studentService")
	private StudentService stuservice;

	// 全部学生信息
	@GetMapping(value = "/query.do")
	public View query() throws Exception
	{
		List<Student> data = stuservice.query();
		if(data != null) {
			return new RestData(data);
		}
		return new RestData("暂无数据！");
	}

	// 增加
	@GetMapping("/insert.do")
	public View insert(String name,Integer sex) throws Exception
	{
		Student s = new Student();
		s.name = name;
		if(name.length() <0 || name.isEmpty())return new RestData("");
		s.sex = sex;
		
		int row = stuservice.insert(s);
		System.out.println("done");
		return new RestData("");
		
	}

	// 根据id删除学生记录
	@GetMapping(value = "/delete.do")
	public View delete(Integer id) throws Exception
	{
		stuservice.delete(id);
		return new RestData("");
	}

	// 修改一条学生信息
	@GetMapping(value = "/update.do")
	public View update(@RequestBody HashMap m) throws Exception
	{
		int row = stuservice.update(m);
		return new RestData("");
	}

	// 根据学号查找一个学生信息
	@GetMapping(value = "/selectID.do")
	public View selectID(int id) throws Exception
	{
		Student student = stuservice.selectID(id);
		return new RestData(student);
	}

	// 根据姓名查找一个学生信息
	@GetMapping(value = "/selectName.do")
	public View selectName(String name) throws Exception
	{
		name = "%" + name + "%";
		List<Student> row = stuservice.selectName(name);
		return new RestData(row);
	}

}
